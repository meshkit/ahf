1function sibhfs = determine_sibling_halffaces_pyramid( nv, elems, varargin) %#codegen
  2%DETERMINE_SIBLING_HALFFACE_PYRAMID Determine the sibling half-face.
  3% DETERMINE_SIBLING_HALFFACE_PYRAMID(NV,ELEMS,SIBHFS) Determines
  4% the sibling half-face.
  5%
  6% SIBHFS = DETERMINE_SIBLING_HALFFACE_PYRAMID(NV,ELEMS)
  7% SIBHFS = DETERMINE_SIBLING_HALFFACE_PYRAMID(NV,ELEMS,SIBHFS)
  8% computes mapping from each half-face to its sibling half-face.
  9%
 10% We assign three bits to local_face_id.
 11%#codegen -args {int32(0), coder.typeof(int32(0), [inf,17],[1,1]),
 12%#codegen coder.typeof(int32(0), [inf,5],[1,1])}
 13
 14% Table for vertices of each face.
 15hf_pyr    = int32([1,4,3,2; 1,2,5,0; 2,3,5,0; 3,4,5,0; 4,1,5,0]);
 16
 17next = int32([2,3,1 0; 2,3,4,1]);
 18prev = int32([3 1 2 0; 4,1,2,3]);
 19
 20%% First, build is_index to store starting position for each vertex.
 21is_index = zeros(nv+1,1, 'int32');
 22nelems = int32(size(elems,1));
 23
 24for ii=1:nelems
 25    if elems(ii,1)==0; nelems=ii-1; break; end
 26
 27    for jj=1:5
 28        nvpf = 3+(jj==1);
 29        vs = elems(ii,hf_pyr(jj,1:nvpf));
 30        v = max( vs, [], 2);
 31        is_index(v+1) = is_index(v+1)+1;
 32    end
 33end
 34is_index(1) = 1;
 35for ii=1:nv; is_index(ii+1) = is_index(ii) + is_index(ii+1); end
 36
 37% v2hf stores mapping from each vertex to half-face ID.
 38% v2oe stores mapping from each vertex to the encoding of the sibling
 39%     edges of each half-face.
 40%v2hf = nullcopy(zeros(is_index(nv+1),1,'int32'));
 41v2hf_cid = nullcopy(zeros(is_index(nv+1),1, 'int32'));
 42v2hf_lfid = nullcopy(zeros(is_index(nv+1),1, 'int8'));
 43
 44v2oe_v1 = nullcopy(zeros(is_index(nv+1),1, 'int32'));
 45v2oe_v2 = nullcopy(zeros(is_index(nv+1),1, 'int32'));
 46
 47for ii=1:nelems
 48    for jj=1:5
 49        nvpf = 3+(jj<4);
 50        vs = elems(ii,hf_pyr(jj,1:nvpf));
 51        [v,kk] = max( vs, [], 2);
 52        
 53        v2oe_v1(is_index(v)) = vs( next(nvpf-2,kk)); 
 54        v2oe_v2(is_index(v)) = vs( prev(nvpf-2,kk));
 55        %v2hf(is_index(v)) = clfids2hfid(ii,jj);
 56        
 57        v2hf_cid(is_index(v)) = ii;
 58        v2hf_lfid(is_index(v)) = jj;
 59
 60        is_index(v) = is_index(v)+1;
 61    end
 62end
 63for ii=nv-1:-1:1; is_index(ii+1) = is_index(ii); end
 64is_index(1)=1;
 65
 66% Fill in sibhfs for each half-face.
 67usestruct=false;
 68if nargin<3 || isempty(varargin{1}) || ~islogical(varargin{1})
 69    sibhfs = zeros(size(elems), 'int32');
 70elseif islogical(varargin{1})
 71    sibhfs = struct( 'cid', zeros(size(elems), 'int32'), ...
 72        'lfid', zeros(size(elems), 'int8'));
 73    usestruct=true;
 74else
 75    sibhfs = varargin{1};
 76    assert( size(sibhfs,1)>=nelems && size(sibhfs,2)==5);
 77    sibhfs(:,:) = int32(0);
 78end
 79
 80for ii=1:nelems
 81    for jj=int32(1):5 % local face ID
 82       if isstruct(sibhfs) && sibhfs.cid(ii,jj) || ...
 83                ~isstruct(sibhfs) && sibhfs(ii,jj); 
 84            continue; 
 85        end
 86        nvpf = 3+(jj==1);
 87        vs = elems(ii, hf_pyr(jj,1:nvpf));  % list of vertices of face
 88        [v,imax] = max( vs, [], 2);
 89
 90        found = false;
 91        v1 = vs(prev(nvpf-2,imax)); v2 = vs(next(nvpf-2,imax));
 92        % Search for sibling half-face.
 93        for index = is_index( v):is_index( v+1)-1
 94            if v2oe_v1(index) == v1 && v2oe_v2(index) == v2 
 95                
 96                if (usestruct)
 97                    sibhfs.lfid(ii,jj) = v2hf_lfid(index);
 98                    sibhfs.cid(ii,jj) = v2hf_cid(index);
 99                    
100                    sibhfs.lfid(v2hf_cid(index),v2hf_lfid(index)) = ii;
101                    sibhfs.cid(v2hf_cid(index),v2hf_lfid(index)) = jj;
102                else
103                    sibhfs(ii,jj) = clfids2hfid(v2hf_cid(index),v2hf_lfid(index));
104                    sibhfs(v2hf_cid(index),v2hf_lfid(index)) = clfids2hfid(ii,jj);
105                end
106                found = true;
107                break;
108            end
109        end
110
111        if ~found
112            for index = is_index( v):is_index( v+1)-1
113                if v2oe_v1(index) == v2 && v2oe_v2(index)==v1 && ...
114                        v2hf_cid(index)~=ii
115                    if nargin==3
116                        error( 'Input mesh is not oriented.');
117                    else
118                        if ~usestruct
119                            sibhfs = zeros(0, 5, 'int32'); return;
120                        else
121                            sibhfs.lfid = zeros(size(elems), 'int8');
122                            sibhfs.fid = zeros(size(elems), 'int32');
123                        end
124                    end
125                end
126            end
127        end
128    end
129end
