1function sibhfs = determine_sibling_halffaces_prism( nv, elems, varargin) %#codegen
  2%DETERMINE_SIBLING_HALFFACE_PRISM Determine the sibling half-face.
  3% DETERMINE_SIBLING_HALFFACE_PRISM(NV,ELEMS,SIBHFS) Determines
  4% the sibling half-face.
  5%
  6% SIBHFS = DETERMINE_SIBLING_HALFFACE_PRISM(NV,ELEMS)
  7% SIBHFS = DETERMINE_SIBLING_HALFFACE_PRISM(NV,ELEMS,SIBHFS)
  8% Computes mapping from each half-face to its sibling half-face.
  9%
 10% We assign three bits to local_face_id.
 11
 12% Note: See http://www.grc.nasa.gov/WWW/cgns/CGNS_docs_current/sids/conv.html for numbering
 13%       convention of faces.
 14%#codegen -args {int32(0), coder.typeof(int32(0), [inf,17],[1,1]),
 15%#codegen coder.typeof(int32(0), [inf,5],[1,1])}
 16
 17% Table for vertices of each face.
 18hf_pri = int32([1,2,5,4; 2,3,6,5; 3,1,4,6; 1 3 2 0; 4 5 6 0]);
 19
 20next = int32([2,3,1 0; 2,3,4,1]);
 21prev = int32([3 1 2 0; 4,1,2,3]);
 22
 23%% First, build is_index to store starting position for each vertex.
 24is_index = zeros(nv+1,1,'int32');
 25nelems = int32(size(elems,1));
 26for ii=1:nelems
 27    if elems(ii,1)==0; nelems=ii-1; break; end
 28
 29    for jj=1:5
 30        nvpf = 3+(jj<4);
 31        vs = elems(ii,hf_pri(jj,1:nvpf));
 32        v = max( vs, [], 2);
 33        is_index(v+1) = is_index(v+1)+1;
 34    end
 35end
 36is_index(1) = 1;
 37for ii=1:nv; is_index(ii+1) = is_index(ii) + is_index(ii+1); end
 38
 39% v2hf stores mapping from each vertex to half-face ID.
 40% v2oe stores mapping from each vertex to the encoding of the sibling
 41%     edges of each half-face.
 42v2hf_cid = nullcopy(zeros(is_index(nv+1),1, 'int32'));
 43v2hf_lfid = nullcopy(zeros(is_index(nv+1),1, 'int8'));
 44
 45v2oe_v1 = nullcopy(zeros(is_index(nv+1),1, 'int32'));
 46v2oe_v2 = nullcopy(zeros(is_index(nv+1),1, 'int32'));
 47
 48for ii=1:nelems
 49    for jj=int32(1):5
 50        nvpf = 3+(jj<4);
 51        vs = elems(ii,hf_pri(jj,1:nvpf));
 52        [v,kk] = max( vs, [], 2);
 53        
 54        v2oe_v1(is_index(v)) = vs( next(nvpf-2,kk)); 
 55        v2oe_v2(is_index(v)) = vs( prev(nvpf-2,kk));
 56        v2hf_cid(is_index(v)) = ii;
 57        v2hf_lfid(is_index(v)) = jj;
 58        is_index(v) = is_index(v)+1;
 59    end
 60end
 61for ii=nv-1:-1:1; is_index(ii+1) = is_index(ii); end
 62is_index(1)=1;
 63
 64% Fill in sibhfs for each half-face.
 65usestruct=false;
 66if nargin<3 || isempty(varargin{1}) || ~islogical(varargin{1})
 67    sibhfs = zeros(size(elems), 'int32');
 68elseif islogical(varargin{1})
 69    sibhfs = struct( 'cid', zeros(size(elems), 'int32'), ...
 70        'lfid', zeros(size(elems), 'int8'));
 71    usestruct=true;
 72else
 73    sibhfs = varargin{1};
 74    assert( size(sibhfs,1)>=nelems && size(sibhfs,2)==5);
 75    sibhfs(:,:) = int32(0);
 76end
 77
 78for ii=1:nelems
 79    for jj=int32(1):5 % local face ID
 80        if isstruct(sibhfs) && sibhfs.cid(ii,jj) || ...
 81                ~isstruct(sibhfs) && sibhfs(ii,jj); 
 82            continue; 
 83        end
 84        nvpf = 3+(jj<4);
 85        vs = elems(ii, hf_pri(jj,1:nvpf));  % list of vertices of face
 86        [v,imax] = max( vs, [], 2);
 87        
 88        found = false;
 89        v1 = vs(prev(nvpf-2,imax)); v2 = vs(next(nvpf-2,imax));
 90        % Search for sibling half-face.
 91        for index = is_index( v):is_index( v+1)-1
 92            if v2oe_v1(index) == v1 && v2oe_v2(index) == v2 
 93                
 94                if ~usestruct
 95                    sibhfs(ii,jj) = clfids2hfid(v2hf_cid(index),v2hf_lfid(index));
 96                    sibhfs(v2hf_cid(index),v2hf_lfid(index)) = clfids2hfid(ii,jj);
 97                else
 98                    sibhfs.lfid(ii,jj) = v2hf_lfid(index);
 99                    sibhfs.cid(ii,jj) = v2hf_cid(index);
100                    
101                    sibhfs.lfid(v2hf_cid(index),v2hf_lfid(index)) = jj;
102                    sibhfs.cid(v2hf_cid(index),v2hf_lfid(index)) = ii;
103                end
104                found = true;
105                break;
106            end
107        end
108
109        if ~found
110            for index = is_index( v):is_index( v+1)-1
111                if v2oe_v1(index) == v2 && v2oe_v2(index)==v1 && ...
112                        v2hf_cid(index)~=ii
113                    if nargin==3
114                        error( 'Input mesh is not oriented.');
115                    else
116                        if ~usestruct
117                             sibhfs = zeros(0, 5, 'int32'); return;
118                        else
119                             sibhfs.cid = zeros(size(elems), 'int32');
120                             sibhfs.lfid = zeros(size(elems), 'int8');   
121                        end
122                    end
123                end
124            end
125        end
126    end
127end
